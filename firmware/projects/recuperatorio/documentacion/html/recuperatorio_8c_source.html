93static void TomarDistancia(void *pvParameter) {
94 uint16_t distancia;
95 uint16_t distanciaaux=0;
96 float aux=0;
97 while (1) {
98 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
99 distancia = HcSr04ReadDistanceInCentimeters();
100 distanciaaux = distancia;
101
102 if(distancia<1000){
103
104 velocidad = (distancia*100)-(distanciaaux*100)/(0,1); // Velocidad en m/s
105
106 SeleccionLED(velocidad);
107
108 if(velocidad>aux)
109 {
110 velocidadmax=velocidad;
111 velocidadmax=aux;
112 }
113
114 }
115
116 if(velocidad==0){
117 vTaskNotifyGiveFromISR(medidaBalanza_task_handle,pdFALSE);
118 }
119
120 else{
121 LedsOffAll();
122 }
123
124 }
125}

153static void MedidaBalanza(void *pvParameter){
154float medida1[50];
155float medida2[50];
156float promedio1 = 0;
157float promedio2 = 0 ;
158 while (1) {
159 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
160 for(uint8_t i=0; i<50; i++){
161 AnalogInputReadSingle(CH1, &medida1[i]);
162 AnalogInputReadSingle(CH2, &medida2[i]);
163 }
164 float total1 = 0;
165 float total2 = 0;
166
167 for(uint8_t i=0; i<50; i++){
168 total1 = total1 + medida1[i];
169 total2 = total2 + medida2[i];
170 }
171 promedio1=total1/50;
172 promedio2=total2/50;
173
174 float medidaGalga1;
175 float medidaGalga2;
176
177 medidaGalga1=(promedio1*1000*20000)/3,3; //convierto de mV a V y luego de V a Kg
178 medidaGalga1=(promedio2*1000*20000)/3,3; //convierto de mV a V y luego de V a Kg
179
180 peso = medidaGalga1+medidaGalga2;
181
182 EnviarNotificacion(peso);
183
184 }
